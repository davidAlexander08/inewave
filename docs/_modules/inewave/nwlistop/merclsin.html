    ]

    @classmethod
    def le_arquivo(
        cls, diretorio: str, nome_arquivo="merclsin.out"

    def escreve_arquivo(self, diretorio: str, nome_arquivo="merclsin.out"):
        self.write(diretorio, nome_arquivo)
